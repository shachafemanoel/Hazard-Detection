function applyDesignSettings(settings) { if (!settings) return; // theme-mode (דוגמה לשינוי משתנים עיקריים) if (settings['theme-mode']) { if (settings['theme-mode'] === 'light') { document.documentElement.style.setProperty('--bg-main', '#f8fafc'); document.documentElement.style.setProperty('--text-primary', '#1a1f2e'); } else { document.documentElement.style.setProperty('--bg-main', '#0a0e1a'); document.documentElement.style.setProperty('--text-primary', '#f8fafc'); } } // כל שאר המשתנים הדינמיים Object.entries(settings).forEach(([key, value]) => { if (key !== 'theme-mode') { document.documentElement.style.setProperty(`--${key}`, value); } }); } // שמירת כל ערכי הטופס ל-localStorage document.getElementById('save-design-btn').addEventListener('click', function() { const form = document.getElementById('design-customizer-form'); const data = {}; Array.from(form.elements).forEach(el => { if (el.name && (el.type !== 'button' && el.type !== 'submit')) { data[el.name] = el.value; } }); localStorage.setItem('design-settings', JSON.stringify(data)); applyDesignSettings(data); // אפשר להחליף ב-toast alert('העיצוב נשמר!'); }); // בטעינת הדף – טען הגדרות קיימות window.addEventListener('DOMContentLoaded', () => { const settings = JSON.parse(localStorage.getItem('design-settings') || '{}'); applyDesignSettings(settings); // עדכן גם את ערכי השדות בטופס Object.entries(settings).forEach(([key, value]) => { const el = document.querySelector(`[name="${key}"]`); if (el) el.value = value; }); });
